""" 공모주 대시보드 — DART 프록시 서버 ==================================== 역할: 브라우저(HTML 대시보드)는 CORS 때문에 DART API를 직접 못 부름. 이 서버가 중간에서 대신 호출하고 파싱해서 결과를 돌려줌. 실행 방법: pip install flask flask-cors requests beautifulsoup4 lxml python dart_proxy.py 실행 후 브라우저 대시보드에서 API 키 입력 → "공모주 가져오기" 클릭 """ import re import json import requests from datetime import datetime, timedelta from flask import Flask, request, jsonify from flask_cors import CORS from bs4 import BeautifulSoup app = Flask(__name__) CORS(app) # 대시보드(HTML)에서 localhost:5001 호출 허용 DART_BASE = "https://opendart.fss.or.kr/api" # ───────────────────────────────────────────── # 유틸: 날짜 문자열 정리 "20250414" → "2025-04-14" # ───────────────────────────────────────────── def fmt_date(s): s = re.sub(r"[^0-9]", "", str(s or "")) if len(s) == 8: return f"{s[:4]}-{s[4:6]}-{s[6:8]}" return "" # ───────────────────────────────────────────── # 유틸: 숫자 문자열 → int "9,500" → 9500 # ───────────────────────────────────────────── def parse_num(s): try: return int(re.sub(r"[^0-9]", "", str(s or "0"))) except: return 0 # ───────────────────────────────────────────── # STEP 1: 공시 목록 조회 # pblntf_ty = "C" → 발행공시 (IPO 증권신고서 포함) # pblntf_detail_ty = "c002" → 증권신고서(지분증권) — 주식 공모 # ───────────────────────────────────────────── def get_ipo_list(api_key, months_ahead=3): today = datetime.today() bgn = (today - timedelta(days=30)).strftime("%Y%m%d") # 1달 전부터 end = (today + timedelta(days=months_ahead * 30)).strftime("%Y%m%d") url = f"{DART_BASE}/list.json" params = { "crtfc_key": api_key, "pblntf_ty": "C", # 발행공시 "pblntf_detail_ty": "c002", # 증권신고서(지분증권) "bgn_de": bgn, "end_de": end, "page_count": 40, "sort": "date", "sort_mth": "desc", } r = requests.get(url, params=params, timeout=10) r.raise_for_status() data = r.json() if data.get("status") != "000": return [], f"DART 오류: {data.get('message','알 수 없음')}" return data.get("list", []), None # ───────────────────────────────────────────── # STEP 2: 공시 원문(XML) 다운로드 후 파싱 # rcept_no (접수번호) 로 원문 URL 얻기 → XML 파싱 # ───────────────────────────────────────────── def get_document_url(api_key, rcept_no): """접수번호 → 문서 목록 → 첫 번째 XML URL 반환""" url = f"{DART_BASE}/document.xml" params = {"crtfc_key": api_key, "rcept_no": rcept_no} r = requests.get(url, params=params, timeout=10) r.raise_for_status() soup = BeautifulSoup(r.content, "lxml-xml") files = soup.find_all("file") # 본문 파일 (확장자 .xml 또는 .htm) for f in files: nm = f.get_text(strip=True) if nm.endswith(".xml") or "본문" in nm: return nm return files[0].get_text(strip=True) if files else None def parse_ipo_document(api_key, rcept_no, corp_name): """ 공시 원문 XML을 파싱해서 공모 정보 추출. DART 증권신고서 XML은 자유형식이라 텍스트 정규식으로 추출. """ result = { "name": corp_name, "subscribeStart": "", "subscribeEnd": "", "refundDate": "", "listingDate": "", "offeringPrice": 0, "offeringPriceRange": "", "offeringShares": 0, "lockupRatio": 0, "floatRatio": 0, "institutionalDemand": 0, "competitionRatio": 0, "manager": "", "market": "KOSDAQ", "sector": "기타", "dart_corp_code": "", } try: # 원문 다운로드 doc_url = f"https://dart.fss.or.kr/dsaf001/main.do?rcpNo={rcept_no}" r = requests.get(doc_url, timeout=10) text = r.text # ── 청약일 ── # 패턴: "청약기간" 뒤에 날짜 m = re.search(r"청약\s*기간[^0-9]*([0-9]{4})[.년\s-]*([0-9]{1,2})[.월\s-]*([0-9]{1,2})[^0-9~∼\-]*[~∼\-]+[^0-9]*([0-9]{4})[.년\s-]*([0-9]{1,2})[.월\s-]*([0-9]{1,2})", text) if m: result["subscribeStart"] = f"{m.group(1)}-{m.group(2).zfill(2)}-{m.group(3).zfill(2)}" result["subscribeEnd"] = f"{m.group(4)}-{m.group(5).zfill(2)}-{m.group(6).zfill(2)}" # ── 환불일 ── m = re.search(r"환불일[^0-9]*([0-9]{4})[.년\s-]*([0-9]{1,2})[.월\s-]*([0-9]{1,2})", text) if m: result["refundDate"] = f"{m.group(1)}-{m.group(2).zfill(2)}-{m.group(3).zfill(2)}" # ── 상장 예정일 ── m = re.search(r"상장\s*예정일[^0-9]*([0-9]{4})[.년\s-]*([0-9]{1,2})[.월\s-]*([0-9]{1,2})", text) if m: result["listingDate"] = f"{m.group(1)}-{m.group(2).zfill(2)}-{m.group(3).zfill(2)}" # ── 공모가 / 공모가 밴드 ── m = re.search(r"확정\s*공모가[^0-9]*([0-9,]+)\s*원", text) if m: result["offeringPrice"] = parse_num(m.group(1)) else: m = re.search(r"공모\s*희망가격[^0-9]*([0-9,]+)\s*[원~∼\-]", text) if m: result["offeringPrice"] = parse_num(m.group(1)) m = re.search(r"공모\s*희망가격[^0-9]*([0-9,]+)\s*[~∼\-]\s*([0-9,]+)\s*원", text) if m: lo = m.group(1).replace(",", "") hi = m.group(2).replace(",", "") result["offeringPriceRange"] = f"{int(lo):,}~{int(hi):,}" # ── 공모 주식 수 ── m = re.search(r"공모\s*주식[^0-9]*([0-9,]+)\s*주", text) if m: result["offeringShares"] = parse_num(m.group(1)) # ── 주관사 ── m = re.search(r"대표\s*주관\s*회사[^\n]*?([가-힣]+증권[^\n,<]*)", text) if m: result["manager"] = m.group(1).strip() # ── KOSPI / KOSDAQ ── if "유가증권시장" in text or "KOSPI" in text: result["market"] = "KOSPI" elif "코스닥" in text or "KOSDAQ" in text: result["market"] = "KOSDAQ" except Exception as e: print(f" [파싱 오류] {corp_name}: {e}") return result # ───────────────────────────────────────────── # STEP 3: 기업 기본 정보 (업종) # ───────────────────────────────────────────── def get_corp_info(api_key, corp_code): url = f"{DART_BASE}/company.json" params = {"crtfc_key": api_key, "corp_code": corp_code} try: r = requests.get(url, params=params, timeout=8) data = r.json() if data.get("status") == "000": return { "sector": data.get("induty", "기타"), "corp_code": corp_code, } except: pass return {} # ───────────────────────────────────────────── # API 엔드포인트: 대시보드 → 이 서버 호출 # ───────────────────────────────────────────── @app.route("/health") def health(): return jsonify({"status": "ok", "message": "DART 프록시 서버 실행 중"}) @app.route("/ipo-list") def ipo_list(): """ 공모주 목록을 DART에서 가져와 파싱 후 반환. Query params: api_key (필수), months (선택, 기본 3) """ api_key = request.args.get("api_key", "") months = int(request.args.get("months", 3)) if not api_key: return jsonify({"status": "error", "message": "api_key 파라미터가 필요합니다"}), 400 print(f"\n[{datetime.now().strftime('%H:%M:%S')}] 공모주 목록 조회 시작...") # 1. 공시 목록 가져오기 filings, err = get_ipo_list(api_key, months) if err: return jsonify({"status": "error", "message": err}), 500 print(f" → 공시 {len(filings)}건 발견") results = [] for i, filing in enumerate(filings[:20]): # 최대 20개 파싱 (속도 고려) corp_name = filing.get("corp_name", "") rcept_no = filing.get("rcept_no", "") corp_code = filing.get("corp_code", "") print(f" [{i+1}/{min(len(filings),20)}] {corp_name} 파싱 중...") # 2. 원문 파싱 ipo_data = parse_ipo_document(api_key, rcept_no, corp_name) # 3. 기업 정보 (업종) corp_info = get_corp_info(api_key, corp_code) ipo_data.update(corp_info) ipo_data["id"] = rcept_no ipo_data["rcept_no"] = rcept_no ipo_data["dart_corp_code"] = corp_code ipo_data["par"] = 500 ipo_data["ticker"] = "" ipo_data["risks"] = [] ipo_data["strengths"] = [] ipo_data["sector_color"] = "#6b7280" ipo_data["description"] = f"DART 공시 자동 수집 (접수번호: {rcept_no})" ipo_data["totalShares"] = 0 # 청약일이 없으면 스킵 (공모주가 아닌 공시일 수 있음) if not ipo_data.get("subscribeStart"): print(f" → 청약일 없음, 스킵") continue results.append(ipo_data) print(f" → 파싱 완료: {len(results)}건\n") return jsonify({ "status": "ok", "count": len(results), "total_filings": len(filings), "list": results, }) @app.route("/dart") def dart_proxy(): """ DART API 직접 프록시 (범용). Query params: path (API 경로), 나머지 파라미터 그대로 전달 예: /dart?path=list.json&crtfc_key=xxx&pblntf_ty=C """ path = request.args.get("path", "list.json") params = {k: v for k, v in request.args.items() if k != "path"} url = f"{DART_BASE}/{path}" try: r = requests.get(url, params=params, timeout=10) return jsonify(r.json()) except Exception as e: return jsonify({"status": "error", "message": str(e)}), 500 if __name__ == "__main__": print("=" * 50) print(" 공모주 대시보드 — DART 프록시 서버") print("=" * 50) print(" 포트: 5001") print(" 상태 확인: http://localhost:5001/health") print(" 공모주 조회: http://localhost:5001/ipo-list?api_key=YOUR_KEY") print("=" * 50) app.run(host="0.0.0.0", port=5001, debug=False)
